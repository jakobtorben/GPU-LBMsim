29 void initialise_simulation(input_struct input, std::string& fname, float& u_lid, float& tau, float& omega)
30 {
31 
32  if (input.Nx % num_threads != 0)
33  throw std::invalid_argument( "Nx must be a multiple of num_threads (32)" );
34 
35  float cs = std::sqrt(1./3.); // lattice speed of sound D2Q9
36  float mach = 0.1; // lattice mach number
37  u_lid = mach * cs; // lid speed
38  float kin_visc = u_lid * float(input.Nx-1) / input.reynolds; // Nx-1 is length of slididng lid
39  tau = (3. * kin_visc + 0.5); // relaxation time, ensures consistency with Navier-Stokes eq
40  omega = 1/tau;
41 
42  // define output filename
43  std::stringstream stream;
44  stream << "LBM_sim_";
45  if (mrt) stream << "MRT"; else stream << "SRT";
46  if (les) stream << "_LES";
47  stream << "_Nx" << input.Nx << "_Ny" << input.Ny << "_Re"<< input.reynolds;
48  fname = stream.str();
49 
50  // print simulation parameters
51  std::cout << "Nx: " << input.Nx << " Ny: " << input.Ny << "\n";
52  std::cout << "Boundary conditions: Lid driven cavity\n";
53  std::cout << "Collision operator: ";
54  if (mrt) std::cout << "MRT"; else std::cout << "SRT";
55  if (les) std::cout << "-LES\n"; else std::cout << "\n";
56  std::cout << "Reynolds number: " << input.reynolds << "\n";
57  std::cout << "kinematic viscosity: " << kin_visc << "\n";
58  std::cout << "u_lid: " << u_lid << "\n";
59  std::cout << "mach number: " << mach << "\n";
60  std::cout << "tau : " << tau << "\n\n";
61 }

117 __global__ void initialise_distributions(int Nx, int Ny, float u_lid, float* f, float* rho_arr, float* ux_arr, float* uy_arr)
118 {
119  int x = blockDim.x*blockIdx.x + threadIdx.x;
120  int y = blockIdx.y;
121 
122  // set density to 1.0 to keep as much precision as possible during calculation
123  rho_arr[arr_idx(Nx, x, y)] = 1.;
124  uy_arr[arr_idx(Nx, x, y)] = 0.;
125  if (y == Ny - 1)
126  ux_arr[arr_idx(Nx, x, y)] = u_lid;
127  else
128  ux_arr[arr_idx(Nx, x, y)] = 0;
129 
130  float ux = ux_arr[arr_idx(Nx, x, y)];
131  float uy = uy_arr[arr_idx(Nx, x, y)];
132  float rho = rho_arr[arr_idx(Nx, x, y)];
133 
134  float uxsq = ux * ux;
135  float uysq = uy * uy;
136  float usq = uxsq + uysq;
137 
138  float uxuy5 = ux + uy;
139  float uxuy6 = -ux + uy;
140  float uxuy7 = -ux - uy;
141  float uxuy8 = ux - uy;
142 
143  float c = 1 - 1.5*usq;
144  float w_rho0 = w0 * rho;
145  float w_rho1 = w1 * rho;
146  float w_rho2 = w2 * rho;
147 
148  f[f_idx_gpu(Nx, Ny, x, y, 0)] = w_rho0*(c );
149  f[f_idx_gpu(Nx, Ny, x, y, 1)] = w_rho1*(c + 3.*ux + c2*uxsq );
150  f[f_idx_gpu(Nx, Ny, x, y, 2)] = w_rho1*(c + 3.*uy + c2*uysq );
151  f[f_idx_gpu(Nx, Ny, x, y, 3)] = w_rho1*(c - 3.*ux + c2*uxsq );
152  f[f_idx_gpu(Nx, Ny, x, y, 4)] = w_rho1*(c - 3.*uy + c2*uysq );
153  f[f_idx_gpu(Nx, Ny, x, y, 5)] = w_rho2*(c + 3.*uxuy5 + c2*uxuy5*uxuy5);
154  f[f_idx_gpu(Nx, Ny, x, y, 6)] = w_rho2*(c + 3.*uxuy6 + c2*uxuy6*uxuy6);
155  f[f_idx_gpu(Nx, Ny, x, y, 7)] = w_rho2*(c + 3.*uxuy7 + c2*uxuy7*uxuy7);
156  f[f_idx_gpu(Nx, Ny, x, y, 8)] = w_rho2*(c + 3.*uxuy8 + c2*uxuy8*uxuy8);
157 }
